Jade

                   h1 Jade - node template engine
                   #container
                       if youAreUsingJade
                           p You are amazing
                       else
                           p Get on it!
                 

                <body>
                   <h1> Jade - node template engine</h1>

                       <p>You are amazing</p>
                   </div>
                </body>

HAML


                %body
                   %h1 HAML - {markup haiku}
                   #container
                       - if youAreUsingHAML
                           %p You are amazing
                       - else
                           %p Get on it!
                 

                <body>
                   <h1> HAML - {markup haiku}</h1>
                   <div id="container">
                       <p>You are amazing</p>
                   </div>
                </body>
